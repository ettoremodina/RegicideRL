 39    def _generate_global_attack_actions(self) -> List[Dict]:
 40        """
 41        Generates all 286 possible unique attack actions in Regicide,
 42        independent of the player's current hand.
 43        """
 44        actions = []
 45        
 46        # 0. Yield
 47        actions.append({"type": "Yield", "cards": []})
 48        
 49        # Generate a standard deck for reference
 50        deck = []
 51        for suit in Suit:
 52            for val in range(1, 14):
 53                deck.append(Card(val, suit))
 54        jester = Card(0, Suit.HEARTS)
 55        
 56        # 1. Single Cards
 57        for card in deck: # All 52 standard cards
 58            actions.append({"type": "Single", "cards": [card]})
 59        actions.append({"type": "Single", "cards": [jester]}) # 1 Jester action
 60        
 61        # 2. Ace + Ace (Animal Companions)
 62        aces = [c for c in deck if c.value == 1]
 63        for combo in itertools.combinations(aces, 2):
 64            actions.append({"type": "Ace+Ace", "cards": list(combo)})
 65            
 66        # 3. Ace + Non-Jester
 67        non_jesters = [c for c in deck if c.value not in (0, 1)]
 68        for ace in aces:
 69            for other in non_jesters:
 70                actions.append({"type": "Ace+Other", "cards": [ace, other]})
 71                
 72        # 4. Same-value combos (sum <= 10)
 73        for val in range(2, 11):
 74            same_val_cards = [c for c in deck if c.value == val]
 75            for r in range(2, 5):
 76                if val * r <= 10:
 77                    for combo in itertools.combinations(same_val_cards, r):
 78                        actions.append({"type": "SameValue", "cards": list(combo)})
 79                        
 80        # Precompute sorted tuples for faster mask generation
 81        for action in actions:
 82            action["sorted_cards"] = tuple(sorted(action["cards"]))
 83            cards = action["cards"]
 84            action["has_jester"] = any(c.value == 0 for c in cards)
 85            action["total_attack"] = sum(c.get_attack_value() for c in cards)
 86            action["played_defense"] = sum(c.get_discard_value() for c in cards)
 87            
 88            suit_vals = set(c.suit.value for c in cards)
 89            action["has_clubs"] = "♣" in suit_vals
 90            action["has_diamonds"] = "♦" in suit_vals
 91            action["has_spades"] = "♠" in suit_vals
 92            action["card_suits"] = list(suit_vals)
 93            
 94        return actions

 96    def get_global_action_mask(
 97        self,
 98        hand: List[Card],
 99        phase: str,
100        game_state: Optional[Dict] = None,
101    ) -> List[int]:
102        """
103        Get a 543-length binary array representing all valid actions globally.
104        Indices 0-285: Attack Actions (Global combinations)
105        Indices 286-541: Defense Actions (Hand-relative subsets)
106        Index 542: Use Solo Jester
107        """
108        mask = np.zeros(GLOBAL_ACTION_SPACE_SIZE, dtype=bool)
109        
110        hand_size = len(hand)
111            
112        if phase == "attack":
113            allow_yield = self._allow_yield(game_state)
114            if allow_yield:
115                mask[0] = 1
116                
117            enemy_attack = 0
118            enemy_health = 0
119            enemy_damage_taken = 0
120            enemy_suit = None
121            jester_cancelled = False
122            
123            if game_state:
124                enemy_attack = game_state.get('enemy_attack', 0)
125                enemy_health = game_state.get('enemy_health', 0)
126                enemy_damage_taken = game_state.get('enemy_damage_taken', 0)
127                enemy_suit = game_state.get('enemy_suit')
128                jester_cancelled = game_state.get('jester_immunity_cancelled', False)
129                
130            hand_defense = sum(c.get_discard_value() for c in hand)
131            
132            has_survivable = False
133            
134            # Find all playable actions first
135            playable_actions = []
136            
137            aces = []
138            non_jesters = []
139            same_val_groups = {}
140            for card in hand:
141                idx = self._attack_action_to_id.get((card,))
142                if idx is not None:
143                    playable_actions.append(idx)
144                    
145                if card.value == 1:
146                    aces.append(card)
147                elif card.value != 0:
148                    non_jesters.append(card)
149                if 2 <= card.value <= 10:
150                    same_val_groups.setdefault(card.value, []).append(card)
151                    
152            if len(aces) >= 2:
153                for c1, c2 in itertools.combinations(aces, 2):
154                    idx = self._attack_action_to_id.get(tuple(sorted((c1, c2))))
155                    if idx is not None:
156                        playable_actions.append(idx)
157                        
158            if aces and non_jesters:
159                for ace in aces:
160                    for other in non_jesters:
161                        idx = self._attack_action_to_id.get(tuple(sorted((ace, other))))
162                        if idx is not None:
163                            playable_actions.append(idx)
164                            
165            for val, cards in same_val_groups.items():
166                n = len(cards)
167                if n >= 2:
168                    for r in range(2, n + 1):
169                        if val * r <= 10:
170                            for combo in itertools.combinations(cards, r):
171                                idx = self._attack_action_to_id.get(tuple(sorted(combo)))
172                                if idx is not None:
173                                    playable_actions.append(idx)
174            
175            if not game_state or enemy_attack == 0:
176                for i in playable_actions:
177                    mask[i] = 1
178            else:
179                for i in playable_actions:
180                    action_info = self._global_attack_actions[i]
181                    cards = action_info["cards"]
182                    
183                    has_jester = action_info["has_jester"]
184                    if has_jester:
185                        mask[i] = 1
186                        has_survivable = True
187                        continue
188                        
189                    total_attack = action_info["total_attack"]
190                    has_clubs = action_info["has_clubs"]
191                    has_diamonds = action_info["has_diamonds"]
192                    has_spades = action_info["has_spades"]
193                    
194                    if not jester_cancelled and enemy_suit in action_info["card_suits"]:
195                        has_clubs = False
196                        has_diamonds = False
197                        has_spades = False
198                        for c in cards:
199                            if c.suit.value != enemy_suit:
200                                val = c.suit.value
201                                if val == "♣": has_clubs = True
202                                elif val == "♦": has_diamonds = True
203                                elif val == "♠": has_spades = True
204                    
205                    if has_diamonds:
206                        mask[i] = 1
207                        has_survivable = True
208                        continue
209                        
210                    total_damage = total_attack * 2 if has_clubs else total_attack
211                    if enemy_damage_taken + total_damage >= enemy_health:
212                        mask[i] = 1
213                        has_survivable = True
214                        continue
215                        
216                    spade_protection = total_attack if has_spades else 0
217                    eff_enemy_attack = max(0, enemy_attack - spade_protection)
218                    
219                    played_defense = action_info["played_defense"]
220                    remaining_defense = hand_defense - played_defense
221                    
222                    if remaining_defense >= eff_enemy_attack:
223                        mask[i] = 1
224                        has_survivable = True
225                        
226                # If no survivable actions (excluding yield), revert to all playable
227                if not has_survivable:
228                    for i in playable_actions:
229                        mask[i] = 1
230                        
231        elif phase == "defense":
232            offset = DEFENSE_ACTION_OFFSET
233            req = game_state.get('enemy_attack', 0) if game_state else 0
234            hand_vals = [c.get_discard_value() for c in hand]
235            
236            sums = [0] * (1 << hand_size)
237            for i in range(hand_size):
238                val = hand_vals[i]
239                bit = 1 << i
240                for b in range(bit):
241                    sums[b | bit] = sums[b] + val
242            
243            has_defense = False
244            for b in range(1, 1 << hand_size):
245                defense = sums[b]
246                if defense >= req:
247                    is_min = True
248                    for i in range(hand_size):
249                        if (b & (1 << i)) and (defense - hand_vals[i] >= req):
250                            is_min = False
251                            break
252                    if is_min:
253                        mask[offset + b] = 1
254                        has_defense = True
255                        
256            if not has_defense:
257                if hand_size > 0:
258                    for b in range(1, 1 << hand_size):
259                        mask[offset + b] = 1
260                else:
261                    mask[0] = 1
262
263        if game_state and game_state.get('can_use_solo_jester', False):
264            mask[SOLO_JESTER_ACTION_ID] = 1
265            
266        # Fallback: MaskablePPO fails drastically if the mask is all zeros
267        if sum(mask) == 0:
268            mask[0] = 1
269            
270        return mask.tolist()

272    def get_all_possible_actions(
273        self,
274        hand: List[Card],
275        phase: str,
276        game_state: Optional[Dict] = None,
277    ) -> List[List[int]]:
278        """
279        Get all possible actions for the current phase
280        Direct computation for fast local mask generation.
281        """
282        local_masks = []
283        hand_size = len(hand)
284        
285        def make_mask(indices):
286            m = [0] * self.max_hand_size
287            for idx in indices:
288                if idx < self.max_hand_size:
289                    m[idx] = 1
290            return m
291            
292        if phase == "attack":
293            allow_yield = self._allow_yield(game_state)
294            if allow_yield:
295                local_masks.append([0] * self.max_hand_size)
296                
297            enemy_attack = 0
298            enemy_health = 0
299            enemy_damage_taken = 0
300            enemy_suit = None
301            jester_cancelled = False
302            
303            if game_state:
304                enemy_attack = game_state.get('enemy_attack', 0)
305                enemy_health = game_state.get('enemy_health', 0)
306                enemy_damage_taken = game_state.get('enemy_damage_taken', 0)
307                enemy_suit = game_state.get('enemy_suit')
308                jester_cancelled = game_state.get('jester_immunity_cancelled', False)
309                
310            hand_defense = sum(c.get_discard_value() for c in hand)
311            
312            playable_indices = []
313            
314            aces_idx = []
315            non_jesters_idx = []
316            same_val_groups = {}
317            for i, card in enumerate(hand):
318                playable_indices.append([i])
319                if card.value == 1:
320                    aces_idx.append(i)
321                elif card.value != 0:
322                    non_jesters_idx.append(i)
323                if 2 <= card.value <= 10:
324                    same_val_groups.setdefault(card.value, []).append(i)
325                    
326            if len(aces_idx) >= 2:
327                for c1, c2 in itertools.combinations(aces_idx, 2):
328                    playable_indices.append([c1, c2])
329                    
330            if aces_idx and non_jesters_idx:
331                for ace_i in aces_idx:
332                    for other_i in non_jesters_idx:
333                        playable_indices.append([ace_i, other_i])
334                        
335            for val, cards_idx in same_val_groups.items():
336                n = len(cards_idx)
337                if n >= 2:
338                    for r in range(2, n + 1):
339                        if val * r <= 10:
340                            for combo in itertools.combinations(cards_idx, r):
341                                playable_indices.append(list(combo))
342                                
343            if not game_state or enemy_attack == 0:
344                for indices in playable_indices:
345                    local_masks.append(make_mask(indices))
346            else:
347                has_survivable = False
348                survivable_masks = []
349                for indices in playable_indices:
350                    cards = [hand[i] for i in indices]
351                    has_jester = any(c.value == 0 for c in cards)
352                    if has_jester:
353                        survivable_masks.append(make_mask(indices))
354                        has_survivable = True
355                        continue
356                        
357                    total_attack = sum(c.get_attack_value() for c in cards)
358                    
359                    has_clubs = False
360                    has_diamonds = False
361                    has_spades = False
362                    
363                    for c in cards:
364                        immune = False if jester_cancelled else (c.suit.value == enemy_suit)
365                        if not immune:
366                            val = c.suit.value
367                            if val == "♣": has_clubs = True
368                            elif val == "♦": has_diamonds = True
369                            elif val == "♠": has_spades = True
370                            
371                    if has_diamonds:
372                        survivable_masks.append(make_mask(indices))
373                        has_survivable = True
374                        continue
375                        
376                    total_damage = total_attack * 2 if has_clubs else total_attack
377                    if enemy_damage_taken + total_damage >= enemy_health:
378                        survivable_masks.append(make_mask(indices))
379                        has_survivable = True
380                        continue
381                        
382                    spade_protection = total_attack if has_spades else 0
383                    eff_enemy_attack = max(0, enemy_attack - spade_protection)
384                    
385                    played_defense = sum(c.get_discard_value() for c in cards)
386                    remaining_defense = hand_defense - played_defense
387                    
388                    if remaining_defense >= eff_enemy_attack:
389                        survivable_masks.append(make_mask(indices))
390                        has_survivable = True
391                        
392                if has_survivable:
393                    local_masks.extend(survivable_masks)
394                else:
395                    for indices in playable_indices:
396                        local_masks.append(make_mask(indices))
397                        
398        elif phase == "defense":
399            req = game_state.get('enemy_attack', 0) if game_state else 0
400            if req <= 0:
401                local_masks.append([0] * self.max_hand_size)
402                return local_masks
403
404            hand_vals = [c.get_discard_value() for c in hand]
405            
406            sums = [0] * (1 << hand_size)
407            for i in range(hand_size):
408                val = hand_vals[i]
409                bit = 1 << i
410                for b in range(bit):
411                    sums[b | bit] = sums[b] + val
412            
413            for b in range(1, 1 << hand_size):
414                defense = sums[b]
415                if defense >= req:
416                    is_min = True
417                    for i in range(hand_size):
418                        if (b & (1 << i)) and (defense - hand_vals[i] >= req):
419                            is_min = False
420                            break
421                    if is_min:
422                        m = [0] * self.max_hand_size
423                        for i in range(hand_size):
424                            if b & (1 << i):
425                                m[i] = 1
426                        local_masks.append(m)
427                        
428        if game_state and game_state.get('can_use_solo_jester', False):
429            local_masks.append([0] * self.max_hand_size + [1])
430            
431        return local_masks

434    def _allow_yield(game_state: Optional[Dict]) -> bool:
435        """Read the current yield flag, including the legacy key name."""
436        if not game_state:
437            return True
438        if "can_yield" in game_state:
439            return game_state["can_yield"]
440        return game_state.get("allow_yield", True)

451    def is_yield_action(self, action_mask: List[int]) -> bool:
452        """Check if an action mask represents a yield action (all zeros)"""
453        return all(x == 0 for x in action_mask)

455    def mask_to_card_indices(self, action_mask: List[int], hand_size: int) -> List[int]:
456        """Convert action mask to list of card indices"""
457        indices = []
458        for i in range(min(len(action_mask), hand_size)):
459            if action_mask[i] == 1:
460                indices.append(i)
461        return indices

463    def cards_to_mask(self, card_indices: List[int]) -> List[int]:
464        """Convert list of card indices to action mask"""
465        action_mask = [0] * self.max_hand_size
466        for idx in card_indices:
467            if idx < self.max_hand_size:
468                action_mask[idx] = 1
469        return action_mask

471    def global_action_to_hand_indices(self, action_id: int, hand: List[Card]) -> List[int]:
472        """
473        Decodes a global action ID (0-542) into a list of hand indices to pass to the env.
474        """
475        if action_id == SOLO_JESTER_ACTION_ID:
476            return [-1]
477            
478        if action_id < 0 or action_id >= GLOBAL_ACTION_SPACE_SIZE:
479            raise ValueError(f"Invalid global action id: {action_id}")
480            
481        offset = DEFENSE_ACTION_OFFSET
482            
483        if action_id < offset:
484            if action_id == 0:
485                return []
486            global_action = self._global_attack_actions[action_id]
487            cards_to_play = global_action["cards"]
488            
489            indices = []
490            used = set()
491            for card in cards_to_play:
492                found = False
493                for i, hand_card in enumerate(hand):
494                    if i not in used and hand_card == card:
495                        indices.append(i)
496                        used.add(i)
497                        found = True
498                        break
499                if not found:
500                    hand_str = ", ".join(str(c) for c in hand)
501                    req_str = ", ".join(str(c) for c in cards_to_play)
502                    raise ValueError(f"Action requires {card} (full action: {req_str}) but it's not in hand or already used. Hand: [{hand_str}]")
503            return indices
504            
505        else:
506            # Defense action
507            val = action_id - offset
508            indices = []
509            for i in range(self.max_hand_size):
510                if val & (1 << i):
511                    indices.append(i)
512            # Ensure indices don't exceed current hand size
513            return [idx for idx in indices if idx < len(hand)]

472    def global_action_to_hand_indices(self, action_id: int, hand: List[Card]) -> List[int]:
473        """
474        Decodes a global action ID (0-542) into a list of hand indices to pass to the env.
475        """
476        if action_id == SOLO_JESTER_ACTION_ID:
477            return [-1]
478            
479        if action_id < 0 or action_id >= GLOBAL_ACTION_SPACE_SIZE:
480            raise ValueError(f"Invalid global action id: {action_id}")
481            
482        offset = DEFENSE_ACTION_OFFSET
483            
484        if action_id < offset:
485            if action_id == 0:
486                return []
487            global_action = self._global_attack_actions[action_id]
488            cards_to_play = global_action["cards"]
489            
490            indices = []
491            used = set()
492            for card in cards_to_play:
493                found = False
494                for i, hand_card in enumerate(hand):
495                    if i not in used and hand_card == card:
496                        indices.append(i)
497                        used.add(i)
498                        found = True
499                        break
500                if not found:
501                    hand_str = ", ".join(str(c) for c in hand)
502                    req_str = ", ".join(str(c) for c in cards_to_play)
503                    raise ValueError(f"Action requires {card} (full action: {req_str}) but it's not in hand or already used. Hand: [{hand_str}]")
504            return indices
505            
506        else:
507            # Defense action
508            val = action_id - offset
509            indices = []
510            for i in range(self.max_hand_size):
511                if val & (1 << i):
512                    indices.append(i)
513            # Ensure indices don't exceed current hand size
514            return [idx for idx in indices if idx < len(hand)]

369                            elif val == "♦": has_diamonds = True
370                            elif val == "♠": has_spades = True
371                            
372                    if has_diamonds:
373                        survivable_masks.append(make_mask(indices))
374                        has_survivable = True
375                        continue
376                        
377                    total_damage = total_attack * 2 if has_clubs else total_attack
378                    if enemy_damage_taken + total_damage >= enemy_health:
379                        survivable_masks.append(make_mask(indices))
380                        has_survivable = True
381                        continue
382                        
383                    spade_protection = total_attack if has_spades else 0
384                    eff_enemy_attack = max(0, enemy_attack - spade_protection)
385                    
386                    played_defense = sum(c.get_discard_value() for c in cards)
387                    remaining_defense = hand_defense - played_defense
388                    
389                    if remaining_defense >= eff_enemy_attack:

391                        has_survivable = True
392                        
393                if has_survivable:
394                    local_masks.extend(survivable_masks)
395                else:
396                    for indices in playable_indices:
397                        local_masks.append(make_mask(indices))
398                        
399        elif phase == "defense":
400            req = game_state.get('enemy_attack', 0) if game_state else 0
401            if req <= 0:
402                local_masks.append([0] * self.max_hand_size)
403                return local_masks
404
405            hand_vals = [c.get_discard_value() for c in hand]
406            
407            sums = [0] * (1 << hand_size)
408            for i in range(hand_size):
409                val = hand_vals[i]
410                bit = 1 << i
411                for b in range(bit):
412                    sums[b | bit] = sums[b] + val
413            
414            for b in range(1, 1 << hand_size):
415                defense = sums[b]
416                if defense >= req:
417                    is_min = True
418                    for i in range(hand_size):
419                        if (b & (1 << i)) and (defense - hand_vals[i] >= req):
420                            is_min = False
421                            break
422                    if is_min:
423                        m = [0] * self.max_hand_size
424                        for i in range(hand_size):
425                            if b & (1 << i):
426                                m[i] = 1
427                        local_masks.append(m)
428                        
429        if game_state and game_state.get('can_use_solo_jester', False):
430            local_masks.append([0] * self.max_hand_size + [1])
431            
432        return local_masks
