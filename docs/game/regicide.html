268        """
269        Play cards and return detailed result information about what happened.
270        Returns a dictionary with the following structure:
271        {
272            'success': bool,          # Whether the action was successful
273            'phase': str,             # Current phase: 'invalid', 'next_player_choice', 'enemy_defeated', 'defense_needed', 'turn_complete'
274            'message': str,           # Description of what happened
275            'enemy_damage': int,      # Damage dealt to enemy (if any)
276            'cards_played': List[str], # String representations of cards played
277            'next_player': int,       # Who should act next (-1 if not applicable)
278            'defense_required': int   # Damage that needs to be defended against (0 if none)
279        }
280        """
281        if self.game_over:
282            return {
283                'success': False,
284                'phase': 'invalid',
285                'message': 'Game is over',
286                'enemy_damage': 0,
287                'cards_played': [],
288                'next_player': -1,
289                'defense_required': 0

293        if not card_indices:
294            return self._handle_yield()

296        player_hand = self.players[self.current_player]
297        
298        # Validate indices
299        if not all(0 <= i < len(player_hand) for i in card_indices):
300            return {
301                'success': False,
302                'phase': 'invalid',
303                'message': 'Invalid card indices',
304                'enemy_damage': 0,
305                'cards_played': [],
306                'next_player': -1,
307                'defense_required': 0
308            }
309        
310        cards_to_play = [player_hand[i] for i in sorted(card_indices, reverse=True)]
311        
312        # Validate combo
313        if not self._is_valid_combo(cards_to_play):

316                'phase': 'invalid',
317                'message': 'Invalid card combination',
318                'enemy_damage': 0,
319                'cards_played': [],
320                'next_player': -1,
321                'defense_required': 0

325        self.last_active_player = self.current_player

327        
328        # Remove cards from hand (keep references in cards_to_play)
329        for i in sorted(card_indices, reverse=True):
330            player_hand.pop(i)

333            cards_played_str = [str(card) for card in cards_to_play]
334            logger.info(f"Player {self.current_player + 1} plays: {', '.join(cards_played_str)}")

430    def _is_valid_combo(cards: List[Card]) -> bool:
431        """Validate a play.
432
433        Jester behaviour intentionally left unchanged per user request (no extra restrictions).
434
435        Animal Companion (Ace, value 1) rules enforced:
436          - Single Ace allowed
437          - Exactly two-card combos allowed: (Ace + Ace) or (Ace + one other card)
438          - Ace cannot participate in larger combos
439        Regular numbered combos (no Aces present):
440          - All cards same value
441          - Total combined value <= 10
442        """
443        if not cards:
444            return False
445        if len(cards) == 1:
446            return True
447
448        aces = [c for c in cards if c.value == 1]
449        others = [c for c in cards if c.value != 1]
450
451        if aces:
452            # Any combo with Ace must be exactly two cards only
453            if len(cards) != 2:
454                return False
455            # Two Aces
456            if len(aces) == 2:
457                return True
458            # One Ace + one other card (rules explicitly forbid Ace + Jester pairing)
459            if len(aces) == 1 and len(others) == 1 and others[0].value != 0:
460                return True
461            return False
462
463        # No Aces: uniform value combo with sum <= 10
464        values = {c.value for c in cards}
465        if len(values) != 1:
466            return False
467        return sum(c.value for c in cards) <= 10

469    def _apply_suit_powers(self, cards: List[Card], total_attack: int):
470        """Apply suit powers with corrected ordering and stacking.
471
472        Adjustments:
473          - Hearts resolved before Diamonds.
474          - Clubs: damage doubled once if any effective club present (not per club).
475          - Spades: each effective spade adds protection (stacks).
476          - Ace pairing (two-card play containing Ace): suit power for each distinct suit counted once.
477          - Immunity: cards matching enemy suit (unless Jester cancelled) skip suit effects.
478        """
479        if not cards or not self.current_enemy:
480            return total_attack
481
482        effective = [c for c in cards if not self._is_immune(c)]
483        immune_cards = [c for c in cards if self._is_immune(c)]
484
485        # Track blocked spade protection for potential Jester retroactive effect
486        has_immune_spade = any(c.suit == Suit.SPADES for c in immune_cards)
487        has_effective_spade = any(c.suit == Suit.SPADES for c in effective)
488        if has_immune_spade and not has_effective_spade:
489            self.blocked_spade_value += total_attack
490
491        if not effective:
492            return total_attack
493
494        suits = {c.suit for c in effective}
495
496        # All suit powers use the base attack value (before clubs doubling)
497        power_value = total_attack
498
499        # Hearts first (heal from discard)
500        if Suit.HEARTS in suits:
501            self._hearts_power(power_value)
502        # Diamonds next (draw cards)
503        if Suit.DIAMONDS in suits:
504            self._diamonds_power(power_value)
505        # Spades protection (cumulative, uses base attack value)
506        if Suit.SPADES in suits:
507            self.current_enemy.spade_protection += power_value
508            self.last_spade_protection_added += power_value
509        # Clubs: doubles DAMAGE only (does not affect other suit power values)
510        if Suit.CLUBS in suits:
511            total_attack *= 2
512
513        return total_attack

520    def _hearts_power(self, value: int):
521        if not self.discard_pile:
522            return
523            
524        if self.deterministic_hearts:
525            # Sort discard pile deterministically (e.g., by value then suit) 
526            # to make perfect information solving possible without random shuffles.
527            self.discard_pile.sort(key=lambda c: (c.value, c.suit.value))
528        else:
529            random.shuffle(self.discard_pile)
530            
531        cards_to_heal = min(value, len(self.discard_pile))
532        healed_cards = self.discard_pile[:cards_to_heal]
533        self.discard_pile = self.discard_pile[cards_to_heal:]
534        # Place healed cards UNDER the tavern deck (bottom). Top is end of list.
535        self.tavern_deck = healed_cards + self.tavern_deck
536        self.last_hearts_healed += cards_to_heal

538    def _diamonds_power(self, value: int):
539        cards_drawn = 0
540        player_idx = self.current_player
541        
542        while cards_drawn < value and self.tavern_deck:
543            if len(self.players[player_idx]) < self.get_max_hand_size():
544                self.players[player_idx].append(self.tavern_deck.pop())
545                cards_drawn += 1
546            
547            player_idx = (player_idx + 1) % self.num_players
548            if player_idx == self.current_player and cards_drawn < value:
549                # Made full circle, check if anyone can still draw
550                if all(len(hand) >= self.get_max_hand_size() for hand in self.players):
551                    break
552        
553        # Sort all hands after drawing cards
554        self._sort_all_hands()
555        self.last_diamonds_drawn += cards_drawn

557    def _defeat_enemy(self):

559        if self.attack_cards_buffer:

569            logger.info(f"Enemy defeated! {self.current_enemy.card} moves to discard.")

214    def clone(self) -> 'Game':
215        """Create a fast deep copy of the game state for search simulation.
216        
217        Manually copies all mutable fields instead of using copy.deepcopy,
218        which is significantly faster for MCTS/PIMC where thousands of
219        clones are created per decision.
220        
221        Returns:
222            A new Game instance with identical but independent state.
223        """
224        # Bypass __init__ and _setup_game entirely
225        new = object.__new__(Game)
226        new.num_players = self.num_players
227        new.deterministic_hearts = self.deterministic_hearts
228        new.current_player = self.current_player
229        new.game_over = self.game_over
230        new.victory = self.victory
231        new.jester_immunity_cancelled = self.jester_immunity_cancelled
232        new.last_active_player = self.last_active_player
233        new.blocked_spade_value = self.blocked_spade_value
234        new.solo_jesters_remaining = self.solo_jesters_remaining
235        new.solo_jesters_used = self.solo_jesters_used
236        
237        # Instrumentation counters
238        new.last_hearts_healed = self.last_hearts_healed
239        new.last_diamonds_drawn = self.last_diamonds_drawn
240        new.last_spade_protection_added = self.last_spade_protection_added
241        
242        # Deep-copy card lists (Card objects are immutable, so shallow-copy of lists suffices)
243        new.players = [hand[:] for hand in self.players]
244        new.tavern_deck = self.tavern_deck[:]
245        new.discard_pile = self.discard_pile[:]
246        new.castle_deck = self.castle_deck[:]
247        new.attack_cards_buffer = self.attack_cards_buffer[:]
248        new.players_yielded_this_round = self.players_yielded_this_round[:]
249        
250        # Clone the current enemy (Enemy is mutable — has damage_taken, spade_protection)
251        if self.current_enemy is not None:
252            new_enemy = object.__new__(Enemy)
253            new_enemy.card = self.current_enemy.card  # Card is immutable
254            new_enemy.health = self.current_enemy.health
255            new_enemy.attack = self.current_enemy.attack
256            new_enemy.damage_taken = self.current_enemy.damage_taken
257            new_enemy.spade_protection = self.current_enemy.spade_protection
258            new.current_enemy = new_enemy
259        else:
260            new.current_enemy = None
261        
262        return new

268    def play_card(self, card_indices: List[int]) -> Dict[str, any]:
269        """
270        Play cards and return detailed result information about what happened.
271        Returns a dictionary with the following structure:
272        {
273            'success': bool,          # Whether the action was successful
274            'phase': str,             # Current phase: 'invalid', 'next_player_choice', 'enemy_defeated', 'defense_needed', 'turn_complete'
275            'message': str,           # Description of what happened
276            'enemy_damage': int,      # Damage dealt to enemy (if any)
277            'cards_played': List[str], # String representations of cards played
278            'next_player': int,       # Who should act next (-1 if not applicable)
279            'defense_required': int   # Damage that needs to be defended against (0 if none)
280        }
281        """
282        if self.game_over:
283            return {
284                'success': False,
285                'phase': 'invalid',
286                'message': 'Game is over',
287                'enemy_damage': 0,
288                'cards_played': [],
289                'next_player': -1,
290                'defense_required': 0
291            }
292        
293        # Handle yield action (empty card_indices)
294        if not card_indices:
295            return self._handle_yield()
296        
297        player_hand = self.players[self.current_player]
298        
299        # Validate indices
300        if not all(0 <= i < len(player_hand) for i in card_indices):
301            return {
302                'success': False,
303                'phase': 'invalid',
304                'message': 'Invalid card indices',
305                'enemy_damage': 0,
306                'cards_played': [],
307                'next_player': -1,
308                'defense_required': 0
309            }
310        
311        cards_to_play = [player_hand[i] for i in sorted(card_indices, reverse=True)]
312        
313        # Validate combo
314        if not self._is_valid_combo(cards_to_play):
315            return {
316                'success': False,
317                'phase': 'invalid',
318                'message': 'Invalid card combination',
319                'enemy_damage': 0,
320                'cards_played': [],
321                'next_player': -1,
322                'defense_required': 0
323            }
324        
325        # Mark this player as having taken an active turn
326        self.last_active_player = self.current_player
327        self._reset_yield_tracking()
328        
329        # Remove cards from hand (keep references in cards_to_play)
330        for i in sorted(card_indices, reverse=True):
331            player_hand.pop(i)
332        
333        if logger.isEnabledFor(logging.INFO):
334            cards_played_str = [str(card) for card in cards_to_play]
335            logger.info(f"Player {self.current_player + 1} plays: {', '.join(cards_played_str)}")
336        else:
337            cards_played_str = []
338
339        # Accumulate attack cards; they will be discarded only once the enemy is defeated
340        self.attack_cards_buffer.extend(cards_to_play)
341        
342        # Handle Jester (multiplayer — solo jesters are separate tokens)
343        if any(card.value == 0 for card in cards_to_play):
344            self.jester_immunity_cancelled = True
345            # Retroactive spade protection: per rules, "spades played prior to
346            # the Jester will begin reducing the attack value of the enemy"
347            if self.blocked_spade_value > 0:
348                self.current_enemy.spade_protection += self.blocked_spade_value
349                self.blocked_spade_value = 0
350            return {
351                'success': True,
352                'phase': 'next_player_choice',
353                'message': 'Jester played! Immunity cancelled. Choose next player.',
354                'enemy_damage': 0,
355                'cards_played': cards_played_str,
356                'next_player': -1,  # Indicates choice needed
357                'defense_required': 0
358            }
359        
360        # Reset instrumentation counters for this play
361        self.last_hearts_healed = 0
362        self.last_diamonds_drawn = 0
363        self.last_spade_protection_added = 0
364        
365        total_attack = sum(card.get_attack_value() for card in cards_to_play)
366        
367        # Apply suit powers
368        total_attack = self._apply_suit_powers(cards_to_play, total_attack)
369        
370        # Deal damage
371        initial_damage = self.current_enemy.damage_taken
372        self.current_enemy.damage_taken += total_attack
373        actual_damage = self.current_enemy.damage_taken - initial_damage
374        if logger.isEnabledFor(logging.INFO):
375            logger.info(f"Enemy takes {actual_damage} damage. (Total HP: {self.current_enemy.health - self.current_enemy.damage_taken}/{self.current_enemy.health})")
376        
377        # Check if enemy defeated
378        if self.current_enemy.is_defeated():
379            self._reset_yield_tracking()
380            self._defeat_enemy()
381            if not self.castle_deck:
382                self.victory = True
383                self.game_over = True
384                return {
385                    'success': True,
386                    'phase': 'victory',
387                    'message': 'Victory! All enemies defeated!',
388                    'enemy_damage': actual_damage,
389                    'cards_played': cards_played_str,
390                    'next_player': -1,
391                    'defense_required': 0
392                }
393            self._check_player_can_act()
394            return {
395                'success': True,
396                'phase': 'enemy_defeated',
397                'message': 'Enemy defeated! Same player continues.',
398                'enemy_damage': actual_damage,
399                'cards_played': cards_played_str,
400                'next_player': self.current_player,
401                'defense_required': 0
402            }
403        else:
404            # Enemy attacks - check if defense is needed
405            enemy_attack = self.current_enemy.get_effective_attack()
406            if logger.isEnabledFor(logging.INFO):
407                logger.info(f"Enemy counter-attacks for {enemy_attack} damage.")
408            if enemy_attack == 0:
409                self._next_player()
410                return {
411                    'success': True,
412                    'phase': 'turn_complete',
413                    'message': 'Enemy attack negated by Spades protection. Turn complete.',
414                    'enemy_damage': actual_damage,
415                    'cards_played': cards_played_str,
416                    'next_player': self.current_player,
417                    'defense_required': 0
418                }
419            else:
420                return {
421                    'success': True,
422                    'phase': 'defense_needed',
423                    'message': f'Enemy attacks for {enemy_attack} damage. Defense required.',
424                    'enemy_damage': actual_damage,
425                    'cards_played': cards_played_str,
426                    'next_player': self.current_player,
427                    'defense_required': enemy_attack
428                }

430    @staticmethod
431    def _is_valid_combo(cards: List[Card]) -> bool:
432        """Validate a play.
433
434        Jester behaviour intentionally left unchanged per user request (no extra restrictions).
435
436        Animal Companion (Ace, value 1) rules enforced:
437          - Single Ace allowed
438          - Exactly two-card combos allowed: (Ace + Ace) or (Ace + one other card)
439          - Ace cannot participate in larger combos
440        Regular numbered combos (no Aces present):
441          - All cards same value
442          - Total combined value <= 10
443        """
444        if not cards:
445            return False
446        if len(cards) == 1:
447            return True
448
449        aces = [c for c in cards if c.value == 1]
450        others = [c for c in cards if c.value != 1]
451
452        if aces:
453            # Any combo with Ace must be exactly two cards only
454            if len(cards) != 2:
455                return False
456            # Two Aces
457            if len(aces) == 2:
458                return True
459            # One Ace + one other card (rules explicitly forbid Ace + Jester pairing)
460            if len(aces) == 1 and len(others) == 1 and others[0].value != 0:
461                return True
462            return False
463
464        # No Aces: uniform value combo with sum <= 10
465        values = {c.value for c in cards}
466        if len(values) != 1:
467            return False
468        return sum(c.value for c in cards) <= 10

470    def _apply_suit_powers(self, cards: List[Card], total_attack: int):
471        """Apply suit powers with corrected ordering and stacking.
472
473        Adjustments:
474          - Hearts resolved before Diamonds.
475          - Clubs: damage doubled once if any effective club present (not per club).
476          - Spades: each effective spade adds protection (stacks).
477          - Ace pairing (two-card play containing Ace): suit power for each distinct suit counted once.
478          - Immunity: cards matching enemy suit (unless Jester cancelled) skip suit effects.
479        """
480        if not cards or not self.current_enemy:
481            return total_attack
482
483        effective = [c for c in cards if not self._is_immune(c)]
484        immune_cards = [c for c in cards if self._is_immune(c)]
485
486        # Track blocked spade protection for potential Jester retroactive effect
487        has_immune_spade = any(c.suit == Suit.SPADES for c in immune_cards)
488        has_effective_spade = any(c.suit == Suit.SPADES for c in effective)
489        if has_immune_spade and not has_effective_spade:
490            self.blocked_spade_value += total_attack
491
492        if not effective:
493            return total_attack
494
495        suits = {c.suit for c in effective}
496
497        # All suit powers use the base attack value (before clubs doubling)
498        power_value = total_attack
499
500        # Hearts first (heal from discard)
501        if Suit.HEARTS in suits:
502            self._hearts_power(power_value)
503        # Diamonds next (draw cards)
504        if Suit.DIAMONDS in suits:
505            self._diamonds_power(power_value)
506        # Spades protection (cumulative, uses base attack value)
507        if Suit.SPADES in suits:
508            self.current_enemy.spade_protection += power_value
509            self.last_spade_protection_added += power_value
510        # Clubs: doubles DAMAGE only (does not affect other suit power values)
511        if Suit.CLUBS in suits:
512            total_attack *= 2
513
514        return total_attack

516    def _is_immune(self, card: Card) -> bool:
517        if self.jester_immunity_cancelled:
518            return False
519        return card.suit == self.current_enemy.card.suit

521    def _hearts_power(self, value: int):
522        if not self.discard_pile:
523            return
524            
525        if self.deterministic_hearts:
526            # Sort discard pile deterministically (e.g., by value then suit) 
527            # to make perfect information solving possible without random shuffles.
528            self.discard_pile.sort(key=lambda c: (c.value, c.suit.value))
529        else:
530            random.shuffle(self.discard_pile)
531            
532        cards_to_heal = min(value, len(self.discard_pile))
533        healed_cards = self.discard_pile[:cards_to_heal]
534        self.discard_pile = self.discard_pile[cards_to_heal:]
535        # Place healed cards UNDER the tavern deck (bottom). Top is end of list.
536        self.tavern_deck = healed_cards + self.tavern_deck
537        self.last_hearts_healed += cards_to_heal
538    
539    def _diamonds_power(self, value: int):
540        cards_drawn = 0
541        player_idx = self.current_player
542        
543        while cards_drawn < value and self.tavern_deck:
544            if len(self.players[player_idx]) < self.get_max_hand_size():
545                self.players[player_idx].append(self.tavern_deck.pop())
546                cards_drawn += 1
547            
548            player_idx = (player_idx + 1) % self.num_players
549            if player_idx == self.current_player and cards_drawn < value:
550                # Made full circle, check if anyone can still draw
551                if all(len(hand) >= self.get_max_hand_size() for hand in self.players):
552                    break
553        
554        # Sort all hands after drawing cards
555        self._sort_all_hands()
556        self.last_diamonds_drawn += cards_drawn

558    def _defeat_enemy(self):
559        # When an enemy is defeated, move all attack-played cards to the discard pile
560        if self.attack_cards_buffer:
561            self.discard_pile.extend(self.attack_cards_buffer)
562            self.attack_cards_buffer = []
563
564        # Exact kill -> enemy card goes to TOP of tavern deck (end of list)
565        if self.current_enemy.damage_taken == self.current_enemy.health:
566            self.tavern_deck.append(self.current_enemy.card)
567            logger.info(f"Enemy exactly defeated! {self.current_enemy.card} joins the Tavern.")
568        else:
569            self.discard_pile.append(self.current_enemy.card)
570            logger.info(f"Enemy defeated! {self.current_enemy.card} moves to discard.")
571
572        # Reset for next enemy
573        if self.castle_deck:
574            self.current_enemy = Enemy(self.castle_deck.pop(0))
575            logger.info(f"New enemy revealed: {self.current_enemy}")
576            self.jester_immunity_cancelled = False
577            self.attack_cards_buffer = []
578            self.blocked_spade_value = 0
579        else:
580            self.current_enemy = None

582    def defend_against_attack(self, cards_to_discard: List[Card]) -> Dict[str, any]:
583        """
584        Defend against enemy attack using specific cards.
585        Returns a dictionary with result information:
586        {
587            'success': bool,          # Whether defense was successful
588            'message': str,           # Description of what happened
589            'defense_value': int,     # Total defense value of discarded cards
590            'damage_blocked': int,    # Amount of damage blocked
591            'cards_discarded': List[str], # String representations of discarded cards
592            'next_player': int,       # Who should act next (-1 if game over)
593            'game_over': bool         # Whether the game ended due to failed defense
594        }
595        """
596        enemy_damage = self.current_enemy.get_effective_attack()
597        
598        if enemy_damage == 0:
599            self._next_player()
600            return {
601                'success': True,
602                'message': 'No damage to defend against',
603                'defense_value': 0,
604                'damage_blocked': 0,
605                'cards_discarded': [],
606                'next_player': self.current_player,
607                'game_over': False
608            }
609        
610        if not cards_to_discard:
611            # No cards provided for defense - game over
612            self.game_over = True
613            return {
614                'success': False,
615                'message': 'No defense provided - game over',
616                'defense_value': 0,
617                'damage_blocked': 0,
618                'cards_discarded': [],
619                'next_player': -1,
620                'game_over': True
621            }
622        
623        # Calculate total defense value
624        defense_value = sum(card.get_discard_value() for card in cards_to_discard)
625        if logger.isEnabledFor(logging.INFO):
626            cards_discarded_str = [str(card) for card in cards_to_discard]
627            logger.info(f"Player {self.current_player + 1} defends with {', '.join(cards_discarded_str)} (value: {defense_value})")
628        else:
629            cards_discarded_str = []
630        
631        # Check if defense is sufficient
632        if defense_value < enemy_damage:
633            self.game_over = True
634            return {
635                'success': False,
636                'message': f'Insufficient defense: {defense_value} < {enemy_damage}',
637                'defense_value': defense_value,
638                'damage_blocked': 0,
639                'cards_discarded': cards_discarded_str,
640                'next_player': -1,
641                'game_over': True
642            }
643        
644        # Remove cards from hand and add to discard pile
645        player_hand = self.players[self.current_player]
646        for card in cards_to_discard:
647            if card in player_hand:
648                player_hand.remove(card)
649                self.discard_pile.append(card)
650        
651        # Sort hand after removing cards
652        self._sort_hand(self.current_player)
653        
654        self._next_player()
655        return {
656            'success': True,
657            'message': f'Defense successful: {defense_value} >= {enemy_damage}',
658            'defense_value': defense_value,
659            'damage_blocked': enemy_damage,
660            'cards_discarded': cards_discarded_str,
661            'next_player': self.current_player,
662            'game_over': False
663        }
