 90    def update_metrics(self, category: str, name: str, value: Any):
 91        """Updates the metric table, saves the metric to disk, and broadcasts to plugins."""
 92        with self.lock:
 93            if category not in self.metrics:
 94                self.metrics[category] = {}
 95            self.metrics[category][name] = value
 96        
 97        # Save locally
 98        self.file_writer.log_metric(category, name, value)
 99        
100        # Broadcast to plugins
101        for plugin in self._plugins:

117    # --- Layout Rendering ---
118    def _update_and_get_layout(self):
119        """Callback for rich.live to constantly fetch the new layout."""
120        # 1. Update Logs
121        with self.lock:
122            queue_snapshot = list(self.log_queue)
123            metrics_snapshot = {k: v.copy() for k, v in self.metrics.items()}
124            
125        self._layout["main_split"]["logs"].update(render_logs_panel(queue_snapshot))
126        
127        # 2. Update Metrics
128        self._layout["main_split"]["sidebar"]["metrics"].update(
129            Panel(render_metrics_table(metrics_snapshot), title="Metrics", border_style="magenta")
130        )
131        
132        # 3. Update Hardware
133        cpu = get_cpu_usage()
134        ram = get_ram_usage()
135        gpus = get_gpu_usage()
136        
137        self._layout["main_split"]["sidebar"]["hardware"].update(
138            Panel(render_hardware_table(cpu, ram, gpus), title="System", border_style="cyan")
139        )
140        
141        # 4. Save Telemetry periodically
142        current_time = time.time()
143        if current_time - self._last_telemetry_time >= self.telemetry_interval:
144            self.file_writer.log_telemetry(cpu, ram, gpus)
145            self._last_telemetry_time = current_time
146        
147        # 5. Update Progress Bar
148        if self.progress_manager.is_active:
149            self._layout["footer"].update(self.progress_manager.render())
150        
151        return self._layout

153    # --- Lifecycle ---
154    def start(self):
155        self._running = True
156        self._last_telemetry_time = time.time() # Reset telemetry timer
157        
158        for plugin in self._plugins:
159            plugin.on_startup(self.config)
160            
161        self._live = Live(get_renderable=self._update_and_get_layout, refresh_per_second=self.refresh_rate, screen=True)
162        self._live.start()
163
164    def stop(self):
165        self._running = False
166        if self._live:
167            self._live.stop()
168            
169        for plugin in self._plugins:
170            plugin.on_shutdown()
